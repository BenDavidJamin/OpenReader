// Filename: router.js
define([
  'jquery',
  'backbone',
  'views/page',
  'app'
], function($, Backbone, Page, App){

  /**
   *
   * @class Router
   * @extends Backbone.Router
   * Provides the routes for the app
   *
   */
  var Router = Backbone.Router.extend({
    /**
     * @method initialize
     * Starts up the Backbone history for correct
     * site navigation.
     */
    initialize: function() {
      Backbone.history.start();
    },

    routes: {
      // Define some URL routes
      "": "index"
    },

    /**
     * @method index
     * The default view for the application.
     */
    index: function(){
        var page = new Page({padding: 20});
        var pageView = page.render().el;
        $("#main").append(pageView);
        App.trigger("calculate-pages");
    }
  });

  return Router;
});
